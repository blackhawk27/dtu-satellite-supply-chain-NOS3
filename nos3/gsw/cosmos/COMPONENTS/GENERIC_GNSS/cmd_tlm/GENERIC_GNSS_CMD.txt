COMMAND GENERIC_GNSS GENERIC_GNSS_GS_PING_CC BIG_ENDIAN "GS->FSW SCIENCE-mode ping; FSW echoes PingSeq via HK LAST_PING_SEQ"
  APPEND_PARAMETER CCSDS_STREAMID      16 UINT MIN_UINT16 MAX_UINT16 0x1952 "CCSDS Packet Identification"
  APPEND_PARAMETER CCSDS_SEQUENCE      16 UINT MIN_UINT16 MAX_UINT16 0xC000 "CCSDS Packet Sequence Control"
  APPEND_PARAMETER CCSDS_LENGTH        16 UINT MIN_UINT16 MAX_UINT16 5      "CCSDS Packet Data Length"
  APPEND_PARAMETER CCSDS_FC            8  UINT MIN_UINT8 MAX_UINT8 4        "CCSDS Command Function Code"
  APPEND_PARAMETER CCSDS_CHECKSUM      8  UINT MIN_UINT8 MAX_UINT8 0        "CCSDS Command Checksum"
  # PING_SEQ is read directly from the C struct on a little-endian FSW host.
  # COSMOS 4.5 ignores the per-parameter LITTLE_ENDIAN override and silently
  # drops WRITE_CONVERSION on cmd write, so callers must byte-swap the value
  # themselves before passing it via cmd("... with PING_SEQ <swapped>"). See
  # gnss_gs_responder.rb / gnss_gs_responder_task.rb for the helper.
  APPEND_PARAMETER PING_SEQ            32 UINT MIN_UINT32 MAX_UINT32 0      "Ground-issued ping sequence number (callers byte-swap)"
